      position: absolute;

    }

      background-color: blue;

  <div id="game">

    <div class="health-bar" id="p2-health">

      <div class="health" id="p2-bar"></div>

    </div>

    <div class="fighter" id="player1"></div>

    <div class="fighter player2" id="player2"></div>

  </div>

  <script>

    const game = document.getElementById("game");

    const p1 = document.getElementById("player1");

    const p2 = document.getElementById("player2");

    const p1Bar = document.getElementById("p1-bar");

    const p2Bar = document.getElementById("p2-bar");

    const gameWidth = window.innerWidth;

    const floor = 0;

    let players = {

      player1: {

        el: p1,

        x: 100,

        y: floor,

        health: 100,

        direction: 1

      },

      player2: {

        el: p2,

        x: gameWidth - 150,

        y: floor,

        health: 100,

        direction: -1

      }

    };

    const speed = 10;

    const jumpHeight = 150;

    const gravity = 10;

    const keys = {};

    document.addEventListener("keydown", (e) => {

      keys[e.key] = true;

    });

    document.addEventListener("keyup", (e) => {

      keys[e.key] = false;

    });

    function update() {

      // Player 1 controls (WASD + F)

      const p1Obj = players.player1;

      if (keys["a"]) p1Obj.x -= speed;

      if (keys["d"]) p1Obj.x += speed;

      if (keys["w"] && p1Obj.y === floor) p1Obj.y = jumpHeight;

      if (keys["f"]) attack(p1Obj, players.player2);

      // Player 2 controls (Arrows + L)

      const p2Obj = players.player2;

      if (keys["ArrowLeft"]) p2Obj.x -= speed;

      if (keys["ArrowRight"]) p2Obj.x += speed;

      if (keys["ArrowUp"] && p2Obj.y === floor) p2Obj.y = jumpHeight;

      if (keys["l"]) attack(p2Obj, players.player1);

      // Apply gravity

      for (let id in players) {

        if (players[id].y > floor) {

          players[id].y -= gravity;

          if (players[id].y < floor) players[id].y = floor;

        }

      }

      // Update DOM positions

      for (let id in players) {

        const f = players[id];

        f.el.style.left = f.x + "px";

        f.el.style.bottom = f.y + "px";

      }

      requestAnimationFrame(update);

    }

    function attack(attacker, defender) {

      const dist = Math.abs(attacker.x - defender.x);

      if (dist < 60) {

        defender.health -= 1;

        updateHealthBars();

        if (defender.health <= 0) {

          alert(`${attacker === players.player1 ? "Player 1" : "Player 2"} wins!`);

          location.reload();

        }

      }

    }

    function updateHealthBars() {

      p1Bar.style.width = players.player1.health + "%";

      p2Bar.style.width = players.player2.health + "%";

    }

    update();

  </script>
